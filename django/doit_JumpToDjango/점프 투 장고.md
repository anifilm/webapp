## Do it! 점프 투 장고


# 프로젝트 만들기

$ mkdir 프로젝트명
$ cd 프로젝트명

$ django-admin startproject mysite .


# 애플리케이션 생성

$ python manage.py startapp polls


# 기본 테이블 생성

$ python manage.py migrate


# 데이터베이스 변경사항 반영

$ python manage.py makemigrations
$ python manage.py migrate


# 프로젝트 시작

$ python manage.py runserver
